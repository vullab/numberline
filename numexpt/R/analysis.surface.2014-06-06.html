RTstats <- data.frame(do.call(rbind, by(dat, dat$subject, function(tmp){
  tmp <- log10(tmp$time)
  return(c(mean(tmp), sd(tmp)))})))
names(RTstats) <- c("RT.mean", "RT.sd")
